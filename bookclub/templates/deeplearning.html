function toggleCheck(element) { console.log("toggleCheck function called"); // Retrieve unit ID and current completion status const unitId = element.getAttribute('data-unit-id'); const isCompleted = element.classList.toggle('checked'); // Toggle class element.innerHTML = isCompleted ? '✓' : '•'; // Update symbol based on completion status // Update the data-completed attribute to reflect the new state element.setAttribute('data-completed', isCompleted); // Send AJAX request to update the completed status in Neo4j fetch(`/toggle_completed_status/`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-CSRFToken': '{{ csrf_token }}', // CSRF token for secure POST request }, body: JSON.stringify({ unit_id: unitId, completed: isCompleted // Pass the updated completed status }) }) .then(response => response.json()) .then(data => { if (data.success) { // Recalculate progress if the AJAX request was successful updateProgress(); } else { alert('Failed to update the status.'); // Revert the change if update failed element.classList.toggle('checked'); element.innerHTML = isCompleted ? '•' : '✓'; } }) .catch(error => { console.error("AJAX request failed:", error); alert('Error updating status.'); }); } // Function to update the progress bar based on checked units function updateProgress() { const totalPages = parseInt(document.getElementById("progress-bar-container").getAttribute("data-total-pages")); let completedPages = 0; // Calculate the number of completed pages by checking checked bullets document.querySelectorAll(".bullet").forEach(bullet => { const pages = parseInt(bullet.getAttribute("data-pages")); const completed = bullet.classList.contains("checked"); if (completed) { completedPages += pages; } }); document.addEventListener("DOMContentLoaded", function() { updateProgress(); // Set initial progress on page load }); // Calculate completion percentage const completionPercentage = (completedPages / totalPages) * 100; // Update the progress bar width and progress text document.getElementById("progress-bar-inner").style.width = completionPercentage + "%"; document.getElementById("progress-text").innerText = `${completedPages}/${totalPages} Pages`; }
